Example Date Graph

#!/usr/bin/gnuplot

reset
set terminal png

set xdata time
set timefmt "%s"
set format x "%M"

set xlabel "Time"
set ylabel "Megs remaining"

set title "Backlog"
set key below
set grid

plot "./short.csv" using 1:2 title "megs"
